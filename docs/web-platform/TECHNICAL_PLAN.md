## Data Schemas

### Property Schema
```typescript
interface Property {
  // Identity
  id: string;
  slug: string;
  name: string;
  tagline: string;
  region: string;                       // e.g., "Quintana Roo, Mexico"

  // Location
  location: {
    address: string;
    city: string;
    state: string;
    country: string;
    coordinates: { lat: number; lng: number };
    distanceFromAirport: string;
    distanceFromJCIHospital: string;
    hospitalName: string;
  };

  // Property Details
  details: {
    askingPrice: number;
    lotSize: string;
    builtArea: string;
    bedrooms: number;
    bathrooms: number;
    yearBuilt: number;
    currentUse: string;
    description: string;
  };

  // Media
  media: {
    heroImage: string;
    gallery: string[];
    virtualTour?: string;
  };

  // Evaluation
  evaluation: {
    totalScore: number;
    maxScore: number;
    passFailGates: {
      name: string;
      status: 'PASS' | 'FAIL' | 'CONDITIONAL';
      notes: string;
    }[];
    categoryScores: {
      category: string;
      score: number;
      maxScore: number;
      notes: string;
    }[];
    strengths: string[];
    concerns: string[];
    openQuestions: string[];
  };

  // Financials
  financials: {
    totalProjectCost: number;
    acquisitionCost: number;
    renovationCost: number;
    softCosts: number;
    costPerRoom: number;
    projections: {
      conservative: ScenarioData;
      base: ScenarioData;
      aggressive: ScenarioData;
    };
    unitEconomics: {
      averageRevenuePerGuest: number;
      cac: number;
      ltv: number;
      paybackMonths: number;
      breakEvenOccupancy: number;
    };
  };

  // Status
  status: 'SHORTLIST' | 'HOLD' | 'PASS';
  statusReason: string;
  recommendation: string;
  nextSteps: string[];

  // Metadata
  lastUpdated: string;
  evaluatedBy: string;
}

interface ScenarioData {
  irr: number;
  npv: number;
  roi: number;
  moic: number;
  revenueByYear: number[];
  ebitdaByYear: number[];
  occupancyByYear: number[];
}
```

### Global Location Schema
```typescript
interface GlobalLocation {
  id: string;
  slug: string;
  name: string;                          // e.g., "Quintana Roo, Mexico"
  country: string;
  countryCode: string;                   // e.g., "MX"
  region: string;                        // e.g., "Americas"
  coordinates: { lat: number; lng: number };

  // Tier & Status
  tier: 1 | 2 | 3 | 4;
  score: number;
  status: 'ACTIVE' | 'PLANNED' | 'RESEARCHING' | 'FUTURE';
  targetYear: number;

  // Regulatory
  regulatory: {
    ibogaineLegal: boolean;
    psilocybinLegal: boolean;
    ayahuascaLegal: boolean;
    notes: string;
  };

  // Strategic
  strategic: {
    targetMarket: string;                // e.g., "US East Coast, Latin America"
    uniqueAdvantage: string;
    challenges: string[];
  };

  // Properties (if any are being evaluated)
  properties?: string[];                 // Property IDs

  // Media
  heroImage: string;
  description: string;
}
```
